Blender Python Script

The automation script used to generate camera movements, orbital animations, and timeline markers for the 2-minute cinematic sequence. Written for Blender 5.1 with camera-fix compatibility.

solar_system_cinematic.py 
"""
Solar System Animation Script — Blender 5.1  (Camera-Fix Edition)
==================================================================
FIXES IN THIS VERSION:
  • SOLAR_CAM renamed to "Camera" to match your actual scene object.
  • setup_overview_camera() uses your exact N-panel coordinates.
  • scene.camera is explicitly set so the Camera appears in the timeline.
  • Marker at frame 1 is force-created directly (bypasses add_cam_marker
    silent-fail bug).
  • Outro marker also force-created the same way.
"""

import bpy
import math
from mathutils import Vector

# ─────────────────────────────────────────────────────────────────
#  SETTINGS
# ─────────────────────────────────────────────────────────────────

FPS           = 24
TOTAL_FRAMES  = 2880          # exactly 2 minutes

INTRO_FRAMES  = 200           # wide solar-system overview
VISIT_FRAMES  = 240           # time parked near each planet
TRANS_FRAMES  = 20            # cut/blend between cameras
OUTRO_FRAMES  = 200           # pull back to wide at the end

# Planet visit order  (must match object names in your scene)
PLANET_ORDER = [
    "Mercury", "Venus", "Earth", "Mars",
    "Jupiter", "Saturn", "Uranus", "Neptune", "Pluto",
]

SOLAR_CAM = "Camera"          # ← matches your actual scene camera name

# ── Camera field-of-view ─────────────────────────────────────────
CAM_FOV   = 50.0   # degrees — applied to every planet camera
SOLAR_FOV = 60.0   # degrees — overview camera

# ── Framing padding ──────────────────────────────────────────────
FRAME_PAD = {
    "Mercury": 1.6,
    "Venus":   1.7,
    "Earth":   1.8,
    "Mars":    1.7,
    "Jupiter": 2.0,
    "Saturn":  2.5,
    "Uranus":  1.9,
    "Neptune": 1.9,
    "Pluto":   1.5,
}

# Camera angle offsets
CAM_OFFSETS = {
    "Mercury": {"elev": 20, "side": 30},
    "Venus":   {"elev": 18, "side": 25},
    "Earth":   {"elev": 22, "side": 35},
    "Mars":    {"elev": 18, "side": 28},
    "Jupiter": {"elev": 25, "side": 40},
    "Saturn":  {"elev": 32, "side": 35},
    "Uranus":  {"elev": 20, "side": 30},
    "Neptune": {"elev": 20, "side": 30},
    "Pluto":   {"elev": 15, "side": 20},
}

# Depth-of-field f-stop per planet
DOF_FSTOP = {
    "Mercury": 2.4, "Venus": 2.8, "Earth": 2.4, "Mars": 2.4,
    "Jupiter": 3.5, "Saturn": 3.5, "Uranus": 2.8, "Neptune": 2.8,
    "Pluto": 2.0,
}

# ─────────────────────────────────────────────────────────────────
#  BLENDER 5.1 COMPATIBLE FCURVE ITERATOR
# ─────────────────────────────────────────────────────────────────

def iter_fcurves(action):
    if hasattr(action, 'layers'):
        for layer in action.layers:
            for strip in layer.strips:
                if hasattr(action, 'slots'):
                    for slot in action.slots:
                        try:
                            cb = strip.channelbag(slot)
                            if cb:
                                yield from cb.fcurves
                        except Exception:
                            pass
                if hasattr(strip, 'fcurves'):
                    yield from strip.fcurves
        return
    if hasattr(action, 'fcurves'):
        yield from action.fcurves

# ─────────────────────────────────────────────────────────────────
#  HELPER UTILITIES
# ─────────────────────────────────────────────────────────────────

def obj(name):
    o = bpy.data.objects.get(name)
    if o is None:
        print(f"  [WARNING] Object '{name}' not found.")
    return o


def goto(frame):
    bpy.context.scene.frame_set(frame)


def world_pos(ob):
    goto(bpy.context.scene.frame_current)
    return ob.matrix_world.translation.copy()


def look_at(cam_ob, target: Vector):
    direction = target - cam_ob.location
    if direction.length < 1e-6:
        return
    cam_ob.rotation_euler = direction.to_track_quat('-Z', 'Y').to_euler()


def kf_loc_rot(ob, frame):
    ob.keyframe_insert(data_path="location",       frame=frame)
    ob.keyframe_insert(data_path="rotation_euler", frame=frame)


def force_cam_marker(cam_ob, frame):
    """Directly create a camera marker — never fails silently."""
    scene = bpy.context.scene
    # Remove any existing marker at this exact frame
    for m in [m for m in scene.timeline_markers if m.frame == frame]:
        scene.timeline_markers.remove(m)
    mk        = scene.timeline_markers.new(name=cam_ob.name, frame=frame)
    mk.camera = cam_ob
    print(f"    [Marker] {cam_ob.name} @ frame {frame}")


def add_cam_marker(cam_name, frame):
    cam_ob = obj(cam_name)
    if cam_ob is None:
        return
    force_cam_marker(cam_ob, frame)


def smooth_cam(cam_ob):
    anim = cam_ob.animation_data
    if anim is None or anim.action is None:
        return
    count = 0
    for fc in iter_fcurves(anim.action):
        for kp in fc.keyframe_points:
            kp.interpolation      = 'BEZIER'
            kp.handle_left_type  = 'AUTO_CLAMPED'
            kp.handle_right_type = 'AUTO_CLAMPED'
        count += len(fc.keyframe_points)
    print(f"    Smoothed {count} keyframe points on {cam_ob.name}")


# ─────────────────────────────────────────────────────────────────
#  PLANET SIZE → CAMERA DISTANCE
# ─────────────────────────────────────────────────────────────────

def get_planet_radius(ob):
    if ob is None:
        return 1.0
    try:
        bbox_world = [ob.matrix_world @ Vector(c) for c in ob.bound_box]
        centre     = sum(bbox_world, Vector()) / 8.0
        radius     = max((c - centre).length for c in bbox_world)
        return max(radius, 0.05)
    except Exception as e:
        print(f"    [WARNING] Could not get radius for {ob.name}: {e}")
        return 1.0


def calc_safe_dist(planet_ob, fov_deg, padding=1.8):
    radius   = get_planet_radius(planet_ob)
    half_fov = math.radians(fov_deg / 2.0)
    dist     = (radius / math.tan(half_fov)) * padding
    return max(dist, radius * 1.05)


def set_cam_fov(cam_ob, fov_deg):
    if cam_ob is None or cam_ob.data is None:
        return
    cam_ob.data.angle = math.radians(fov_deg)


# ─────────────────────────────────────────────────────────────────
#  OFFSET VECTOR
# ─────────────────────────────────────────────────────────────────

def cam_offset_vector(planet_pos: Vector, dist, elev_deg, side_deg):
    elev = math.radians(elev_deg)
    side = math.radians(side_deg)

    sun_to_planet = planet_pos.copy()
    sun_to_planet.z = 0
    if sun_to_planet.length > 1e-6:
        sun_to_planet.normalize()
    else:
        sun_to_planet = Vector((1, 0, 0))

    perp     = Vector((-sun_to_planet.y, sun_to_planet.x, 0))
    radial   = sun_to_planet * math.cos(elev) * math.cos(side)
    lateral  = perp          * math.cos(elev) * math.sin(side)
    vertical = Vector((0, 0, math.sin(elev)))

    return planet_pos + (radial + lateral + vertical).normalized() * dist


# ─────────────────────────────────────────────────────────────────
#  STEP 1 — SCENE SETUP
# ─────────────────────────────────────────────────────────────────

def setup_scene():
    scene = bpy.context.scene
    scene.frame_start = 1
    scene.frame_end   = TOTAL_FRAMES
    scene.render.fps  = FPS
    scene.render.resolution_x          = 1920
    scene.render.resolution_y          = 1080
    scene.render.resolution_percentage = 100
    print(f"[Scene] {TOTAL_FRAMES} frames @ {FPS} fps = {TOTAL_FRAMES / FPS:.1f} sec")


# ─────────────────────────────────────────────────────────────────
#  STEP 2 — WIDE OVERVIEW CAMERA  (intro & outro)
# ─────────────────────────────────────────────────────────────────

def setup_overview_camera():
    cam = obj(SOLAR_CAM)
    if cam is None:
        return

    set_cam_fov(cam, SOLAR_FOV)

    # ── Exact coordinates from your N panel ──────────────────────
    start_loc = Vector((27.923, -203.09, 87.105))
    start_rot = (math.radians(65.574), math.radians(2.5453), math.radians(6.8489))

    # Gentle zoom-in during intro (same angle, slightly closer)
    end_loc = Vector((18.0, -130.0, 56.0))
    end_rot = start_rot

    # Outro pulls back to starting position
    outro_loc = Vector((27.923, -203.09, 87.105))
    outro_rot = start_rot

    # --- INTRO -------------------------------------------------------
    goto(1)
    cam.location       = start_loc
    cam.rotation_euler = start_rot
    kf_loc_rot(cam, 1)

    goto(INTRO_FRAMES)
    cam.location       = end_loc
    cam.rotation_euler = end_rot
    kf_loc_rot(cam, INTRO_FRAMES)

    # --- OUTRO -------------------------------------------------------
    outro_start = TOTAL_FRAMES - OUTRO_FRAMES
    goto(outro_start)
    cam.location       = end_loc
    cam.rotation_euler = end_rot
    kf_loc_rot(cam, outro_start)

    goto(TOTAL_FRAMES)
    cam.location       = outro_loc
    cam.rotation_euler = outro_rot
    kf_loc_rot(cam, TOTAL_FRAMES)

    smooth_cam(cam)

    # ── Make this the active scene camera & add frame-1 marker ───
    bpy.context.scene.camera = cam
    force_cam_marker(cam, 1)

    print(f"[{SOLAR_CAM}] Overview camera set. Scene camera = {cam.name}")


# ─────────────────────────────────────────────────────────────────
#  STEP 3 — PLANET CAMERA VISITS
# ─────────────────────────────────────────────────────────────────

def setup_planet_visits():
    scene = bpy.context.scene

    # Remove ALL existing camera markers so we start clean
    for m in [m for m in scene.timeline_markers if m.camera is not None]:
        scene.timeline_markers.remove(m)

    # Re-add the frame-1 overview marker after clearing
    cam_ob = obj(SOLAR_CAM)
    if cam_ob:
        force_cam_marker(cam_ob, 1)

    current = INTRO_FRAMES + 1

    for planet_name in PLANET_ORDER:
        cam_name = planet_name + "_Camera"
        cam      = obj(cam_name)
        planet   = obj(planet_name)

        if cam is None or planet is None:
            print(f"  [SKIP] {planet_name} — camera or planet not found.")
            current += VISIT_FRAMES + TRANS_FRAMES
            continue

        set_cam_fov(cam, CAM_FOV)

        offsets = CAM_OFFSETS[planet_name]
        elev    = offsets["elev"]
        side    = offsets["side"]
        pad     = FRAME_PAD[planet_name]

        # ── Arrival ──────────────────────────────────────────────
        arr_frame = current
        goto(arr_frame)
        bpy.context.view_layer.update()
        p_pos    = planet.matrix_world.translation.copy()
        dist     = calc_safe_dist(planet, CAM_FOV, padding=pad)
        arr_loc  = cam_offset_vector(p_pos, dist * 1.35, elev + 10, side - 20)
        cam.location = arr_loc
        look_at(cam, p_pos)
        kf_loc_rot(cam, arr_frame)

        # ── Mid arc ──────────────────────────────────────────────
        mid_frame = current + VISIT_FRAMES // 2
        goto(mid_frame)
        bpy.context.view_layer.update()
        p_pos_mid = planet.matrix_world.translation.copy()
        dist_mid  = calc_safe_dist(planet, CAM_FOV, padding=pad)
        mid_loc   = cam_offset_vector(p_pos_mid, dist_mid, elev, side + 15)
        cam.location = mid_loc
        look_at(cam, p_pos_mid)
        kf_loc_rot(cam, mid_frame)

        # ── Departure ────────────────────────────────────────────
        dep_frame = current + VISIT_FRAMES
        goto(dep_frame)
        bpy.context.view_layer.update()
        p_pos_dep = planet.matrix_world.translation.copy()
        dist_dep  = calc_safe_dist(planet, CAM_FOV, padding=pad)
        dep_loc   = cam_offset_vector(p_pos_dep, dist_dep * 1.2, elev - 8, side + 40)
        cam.location = dep_loc
        look_at(cam, p_pos_dep)
        kf_loc_rot(cam, dep_frame)

        force_cam_marker(cam, arr_frame)
        smooth_cam(cam)

        radius = get_planet_radius(planet)
        print(f"  [{planet_name}]  radius={radius:.2f}  dist={dist:.2f}  "
              f"frames {arr_frame}–{dep_frame}")

        current = dep_frame + TRANS_FRAMES

    # Outro — switch back to overview camera
    outro_start = TOTAL_FRAMES - OUTRO_FRAMES
    cam_ob = obj(SOLAR_CAM)
    if cam_ob:
        force_cam_marker(cam_ob, outro_start)
    print(f"[Outro] {SOLAR_CAM} resumes @ frame {outro_start}")
    print(f"[Timeline] Final frame used: {current} / {TOTAL_FRAMES}")


# ─────────────────────────────────────────────────────────────────
#  STEP 4 — DEPTH OF FIELD
# ─────────────────────────────────────────────────────────────────

def setup_dof():
    for planet_name in PLANET_ORDER:
        cam_name = planet_name + "_Camera"
        cam_ob   = obj(cam_name)
        planet   = obj(planet_name)
        if cam_ob is None or planet is None:
            continue
        cam_data = cam_ob.data
        cam_data.dof.use_dof        = True
        cam_data.dof.focus_object   = planet
        cam_data.dof.aperture_fstop = DOF_FSTOP.get(planet_name, 2.8)
        print(f"  [DoF] {cam_name} → focus {planet_name}  f/{DOF_FSTOP[planet_name]}")
    print("[DoF] All planet cameras configured.")


# ─────────────────────────────────────────────────────────────────
#  STEP 5 — EXTEND ORBIT ANIMATIONS
# ─────────────────────────────────────────────────────────────────

def extend_orbit_animations():
    orbit_names = [p + "_Orbit" for p in PLANET_ORDER]
    for orbit_name in orbit_names:
        orbit_ob = bpy.data.objects.get(orbit_name)
        if orbit_ob is None:
            continue
        anim_data = orbit_ob.animation_data
        if anim_data is None:
            continue
        for track in anim_data.nla_tracks:
            for strip in track.strips:
                length = max((strip.frame_end - strip.frame_start), 1)
                strip.repeat        = TOTAL_FRAMES / length
                strip.frame_end     = TOTAL_FRAMES
                strip.extrapolation = 'HOLD'
                print(f"  [NLA] {orbit_name} → extended to frame {TOTAL_FRAMES}")
    print("[NLA] Orbit animation strips extended.")


# ─────────────────────────────────────────────────────────────────
#  STEP 6 — TIMELINE REPORT
# ─────────────────────────────────────────────────────────────────

def print_timeline_report():
    current = INTRO_FRAMES + 1
    print("\n" + "─" * 60)
    print("  TIMELINE REPORT")
    print("─" * 60)
    print(f"  Frames 1–{INTRO_FRAMES}  : Intro  ({SOLAR_CAM})")
    for planet_name in PLANET_ORDER:
        end    = current + VISIT_FRAMES
        secs_s = current / FPS
        secs_e = end     / FPS
        planet = bpy.data.objects.get(planet_name)
        r      = get_planet_radius(planet) if planet else 0
        goto(current)
        if planet:
            bpy.context.view_layer.update()
        dist = calc_safe_dist(planet, CAM_FOV, FRAME_PAD[planet_name]) if planet else 0
        print(f"  Frames {current:4d}–{end:4d} : {planet_name:<8s}  "
              f"({secs_s:.1f}s–{secs_e:.1f}s)  "
              f"radius={r:.2f}  cam_dist≈{dist:.2f}")
        current = end + TRANS_FRAMES
    outro_start = TOTAL_FRAMES - OUTRO_FRAMES
    print(f"  Frames {outro_start}–{TOTAL_FRAMES} : Outro   ({SOLAR_CAM})")
    print(f"\n  Total: {TOTAL_FRAMES} frames = {TOTAL_FRAMES / FPS:.1f} seconds")
    print("─" * 60 + "\n")
    goto(1)


# ─────────────────────────────────────────────────────────────────
#  MAIN
# ─────────────────────────────────────────────────────────────────

def main():
    print("\n" + "=" * 60)
    print("  SOLAR SYSTEM ANIMATION — Blender 5.1  (Camera-Fix)")
    print("=" * 60 + "\n")

    setup_scene()
    extend_orbit_animations()
    setup_overview_camera()    # sets scene.camera + frame-1 marker
    setup_planet_visits()      # planet camera markers
    setup_dof()
    print_timeline_report()

    goto(1)

    print("=" * 60)
    print("  DONE!  Press SPACE to preview, Ctrl+F12 to render.")
    print(f"  Duration: {TOTAL_FRAMES} frames = {TOTAL_FRAMES / FPS:.1f} sec @ {FPS} fps")
    print("=" * 60 + "\n")


main()
